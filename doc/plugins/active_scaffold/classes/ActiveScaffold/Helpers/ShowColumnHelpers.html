 5:       def show_column_value(record, column)
 6:         # check for an override helper
 7:         if show_column_override? column
 8:           # we only pass the record as the argument. we previously also passed the formatted_value,
 9:           # but mike perham pointed out that prohibited the usage of overrides to improve on the

11:           send(show_column_override(column), record)
12:         # second, check if the dev has specified a valid list_ui for this column

15:         else
16:           if column.column and override_show_column_ui?(column.column.type)
17:             send(override_show_column_ui(column.column.type), column, record)
18:           else
19:             get_column_value(record, column)
20:           end
21:         end
