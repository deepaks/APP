link_to_visibility_toggle(options = {})

Creates a javascript-based link that toggles the visibility of some element on the page. By default, it toggles the visibility of the sibling after the one it‘s nested in. You may pass custom javascript logic in options[:of] to change that, though. For example, you could say :of => ’$("my_div_id")’. You may also flag whether the other element is visible by default or not, and the initial text will adjust accordingly.

130:         url_options = url_options.clone
131:         url_options[:action] = link.action
132:         url_options[:controller] = link.controller if link.controller
133:         url_options.delete(:search) if link.controller and link.controller.to_s != params[:controller]
134:         url_options.merge! link.parameters if link.parameters
135: 
136:         html_options = link.html_options.merge({:class => link.action})
137:         if link.inline?
138:           # NOTE this is in url_options instead of html_options on purpose. the reason is that the client-side
139:           # action link javascript needs to submit the proper method, but the normal html_options[:method]
140:           # argument leaves no way to extract the proper method from the rendered tag.
141:           url_options[:_method] = link.method
142: 
143:           if link.method != :get and respond_to?(:protect_against_forgery?) and protect_against_forgery?
144:             url_options[:authenticity_token] = form_authenticity_token
145:           end
146: 
147:           # robd: protect against submitting get links as forms, since this causes annoying 
148:           # 'Do you wish to resubmit your form?' messages whenever you go back and forwards.
149:         elsif link.method != :get
150:           # Needs to be in html_options to as the adding _method to the url is no longer supported by Rails
151:           html_options[:method] = link.method
152:         end
153: 
154:         html_options[:confirm] = link.confirm if link.confirm?
155:         html_options[:position] = link.position if link.position and link.inline?
156:         html_options[:class] += ' action' if link.inline?
157:         html_options[:popup] = true if link.popup?
158:         html_options[:id] = action_link_id("#{url_options[:parent_controller] + '_' if url_options[:parent_controller]}" + url_options[:action],url_options[:id] || url_options[:parent_id])

160:         if link.dhtml_confirm?
161:           html_options[:class] += ' action' if !link.inline?
162:           html_options[:page_link] = 'true' if !link.inline?
163:           html_options[:dhtml_confirm] = link.dhtml_confirm.value
164:           html_options[:onclick] = link.dhtml_confirm.onclick_function(controller,action_link_id(url_options[:action],url_options[:id] || url_options[:parent_id]))
165:         end
166:         html_options[:class] += " #{link.html_options[:class]}" unless link.html_options[:class].blank?
167: 
168:         # issue 260, use url_options[:link] if it exists. This prevents DB data from being localized.
